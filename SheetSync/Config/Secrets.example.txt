import Foundation

/// Copy this file to Secrets.swift and fill in your Google OAuth credentials.
/// See docs/GOOGLE_SETUP.md for detailed instructions.
///
/// IMPORTANT: Secrets.swift is gitignored and should never be committed.
///
/// SETUP STEPS:
/// 1. Create an iOS OAuth client in Google Cloud Console
/// 2. Set Bundle ID to: com.gsheetsync.app
/// 3. Copy your Client ID below
/// 4. Update Info.plist URL scheme to: com.googleusercontent.apps.{YOUR_CLIENT_ID_PREFIX}
///    (The prefix is everything before .apps.googleusercontent.com)
enum Secrets {
    /// Your Google OAuth 2.0 Client ID from an iOS OAuth client
    /// Format: "123456789-xxxx.apps.googleusercontent.com"
    static let googleClientId = "YOUR_CLIENT_ID.apps.googleusercontent.com"

    /// The redirect scheme is automatically derived from your client ID
    /// Format: com.googleusercontent.apps.{CLIENT_ID_PREFIX}
    static var googleRedirectScheme: String {
        let prefix = googleClientId.replacingOccurrences(of: ".apps.googleusercontent.com", with: "")
        return "com.googleusercontent.apps.\(prefix)"
    }

    /// The full redirect URI used in OAuth flow
    static var googleRedirectURI: String {
        "\(googleRedirectScheme):/oauth2callback"
    }
}
